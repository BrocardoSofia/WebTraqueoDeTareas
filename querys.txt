existeEmail(p_email) 
{
   let resp;
   const exist = await sequelize.query ("SELECT 1 
                                         FROM Usuarios AS u
                                         WHERE u.email = " + p_email);
   try
   {
      if(exist)
      {
        res.json({respuesta: true})
      }
      else{
         res.status(404).json({respuesta : false})
      }
    }catch (error){
       res.json({msg: ''})
    }

}
